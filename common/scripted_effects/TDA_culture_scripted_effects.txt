#Adds cultures to the array and sorts it
STATE_sort_culture_array = {
	#Either doesn't have an array or the array is 100% other
	if = {
		limit = { 
			OR = {
				NOT = { has_variable  = culture_index_array^num } 
				AND = {
					has_variable = culture_index_array^num				#Has array, is one long, and that one is 'other'
					check_variable = { culture_index_array^num = 1 }
					check_variable = { culture_value_array^0 = 0 }
				}
			}
		}
		
		resize_array = { array = culture_index_array size = 1 value = 0 }
		resize_array = { array = culture_value_array size = 1 value = 0 }
		
		set_variable = { culture_index_array^0 = 0 }
		set_variable = { culture_value_array^0 = 100 }
	}
	#Array exists - make sure the variables add up to 100 then sort them
	else = {
		#Create new arrays to hold our variables
		clear_array = culture_index_array_HOLD
		clear_array = culture_value_array_HOLD
		
		for_loop_effect = {
			end = culture_index_array^num
			break = break1
			value = v1
			
			if = {
				limit = { is_in_array = { array = culture_index_array_HOLD value = culture_index_array^v1 } }		#Is the current index value already in the array? (Do we have the same culture twice)
				
				#Find the repeated culture and add to it
				for_loop_effect = {
					end = culture_index_array_HOLD^num
					break = break2
					value = v2
					
					if = {
						limit = { check_variable = { culture_index_array_HOLD^v2 = culture_index_array^v1 } }
						add_to_variable = { culture_value_array_HOLD^v2 = culture_value_array^v1 }
						set_temp_variable = { break2 = 1 }
					}
				}
			}
			else = {	#Otherwise add it to our array
				add_to_array = { culture_index_array_HOLD = culture_index_array^v1 }
				add_to_array = { culture_value_array_HOLD = culture_value_array^v1 }
			}
		}
			
		#Get the sum of the array
		set_variable = { total_count = 0 }
		for_each_loop = {
			array = culture_value_array_HOLD
			value = v

			add_to_variable = { total_count = v }
		}
		
		#If not equal to 100 multiply each value by (100 / total)
		if = { 
			limit = { NOT = { check_variable = { total_count = 100 } } } 
			set_temp_variable = { multiplier = 100 }
			divide_temp_variable = { multiplier = total_count }
			
			for_loop_effect = {
				end = culture_value_array_HOLD^num
				value = v
				
				multiply_variable = { culture_value_array_HOLD^v = multiplier }
			}
		}
		
		clear_variable = total_count
		
		
		#We now have our array - now we sort it
		resize_array = { array = arr size = culture_value_array_HOLD^num value = 0 }
		for_each_loop = {
			array = culture_value_array_HOLD
			value = v
			index = i
			
			set_variable = { arr^i = v }
		}
		
		#Sort array 'arr'
		set_variable = { high = arr^num }
		subtract_from_variable = { high = 1 }
		set_variable = { low = 0 }
		
		STATE_sort_culture_quicksort = yes
		
		clear_variable = high
		clear_variable = low
		
#		log = "Standard:"
#		for_each_loop = {
#			array = culture_index_array
#			value = v
#			index = i
#			
#			log = "[?v] : [?culture_value_array^i]"
#		}
#		
#		log = " "
#		log = "Hold:"
#		for_each_loop = {
#			array = culture_index_array_HOLD
#			value = v
#			index = i
#			
#			log = "[?v] : [?culture_value_array_HOLD^i]"
#		}
#		
#		log = " "
#		log = "arr:"
#		for_each_loop = {
#			array = arr
#			value = v
#			index = i
#			
#			log = "[?v]"
#		}
#	
#		log = " "
		
		#We now have 5 arrays
		clear_array = culture_index_array
		clear_array = culture_value_array
		
		resize_array = { array = culture_index_array value = 0 size = culture_index_array_HOLD^num }
		resize_array = { array = culture_value_array value = 0 size = culture_value_array_HOLD^num }
	
		#Go over 'arr' (the ordered array) and set the values of culture_index_array and culture_value_array
		#Used a linear search instead of binary search as the ethnicity lists are very small
		for_each_loop = {
			array = arr
			value = v
			index = i
			break = break
			
			for_loop_effect = {
				end = culture_value_array_HOLD^num
				value = i2
				break = break2
				
#				log = "[?i], [?i2]: [?v], [?culture_value_array_HOLD^i2]"

				if = {
					limit = { check_variable = { culture_value_array_HOLD^i2 = v } }
					set_variable = { culture_index_array^i = culture_index_array_HOLD^i2 }
					set_variable = { culture_value_array^i = culture_value_array_HOLD^i2 }
					set_temp_variable = { break2 = 1 }
				}
			}
		}
		
		clear_array = arr
		clear_array = culture_index_array_HOLD
		clear_array = culture_value_array_HOLD
	}
}

#Code taken from Yard1
STATE_sort_culture_quicksort = {
	# set up the auxilary stack
	set_temp_variable = { stack_size = high }
	subtract_from_temp_variable = { stack_size = low }
	add_to_temp_variable = { stack_size = 1 }
	resize_temp_array = {
		array = stack
		value = 0
		size = stack_size
	}

	set_temp_variable = { top = 0 }
	set_temp_variable = { stack^top = low }
	add_to_temp_variable = { top = 1 }
	set_temp_variable = { stack^top = high }

	while_loop_effect = {
		limit = { 
			check_variable = {
				var = top
				value = 0
				compare = greater_than_or_equals
			}
		}
		set_variable = { high = stack^top }
		subtract_from_temp_variable = { top = 1 }
		set_variable = { low = stack^top }
		subtract_from_temp_variable = { top = 1 }

		# Get random pivot
		set_variable = { random_pivot = random }
		multiply_variable = { random_pivot = high }
		round_variable = random_pivot
		add_to_variable = { random_pivot = low }
		clamp_variable = {
			var = random_pivot
			min = low
			max = high
		}
		set_temp_variable = { swap = arr^high }
		set_variable = { arr^high = arr^random_pivot }
		set_variable = { arr^random_pivot = swap }

		set_temp_variable = { idx = low }
		subtract_from_temp_variable = { idx = 1 }

		set_temp_variable = { min = arr^high }
	  
		for_loop_effect = {
			start = low
			end = high
			if = {
				limit = {
					check_variable = {
						var = arr^v
						value = min
						compare = greater_than_or_equals
					}
				}
				add_to_temp_variable = { idx = 1 }
				set_temp_variable = { swap = arr^idx }
				set_variable = { arr^idx = arr^v }
				set_variable = { arr^v = swap }
			}
		}

		add_to_temp_variable = { idx = 1 }
		set_temp_variable = { swap = arr^idx }
		set_variable = { arr^idx = arr^high }
		set_variable = { arr^high = swap }

		# we return the index
		set_variable = { qs_partition_return = idx }

		set_temp_variable = { qs_partition_return_plus_1 = qs_partition_return }
		set_temp_variable = { qs_partition_return_minus_1 = qs_partition_return }
		add_to_temp_variable = { qs_partition_return_plus_1 = 1 } 
		subtract_from_temp_variable = { qs_partition_return_minus_1 = 1 } 

		if = {
			limit = { check_variable = { qs_partition_return_minus_1 > low } }
			add_to_temp_variable = { top = 1 }
			set_temp_variable = { stack^top = low }
			add_to_temp_variable = { top = 1 }
			set_temp_variable = { stack^top = qs_partition_return_minus_1 }
		}

		if = {
			limit = { check_variable = { qs_partition_return_plus_1 < high } }
			add_to_temp_variable = { top = 1 }
			set_temp_variable = { stack^top = qs_partition_return_plus_1 }
			add_to_temp_variable = { top = 1 }
			set_temp_variable = { stack^top = high }
		}
	}

	# clear "temporary" variables
	clear_variable = qs_partition_return
	clear_variable = random_pivot
}
