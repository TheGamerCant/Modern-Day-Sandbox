	text = {
		trigger = {
			NOT = { has_global_flag = TDA_using_super_cultures_flag }
			check_variable = { var = FROM.culture_value_array^0 value = 100 compare = greater_than_or_equals }
			
			set_temp_variable = { dem1 = FROM.culture_index_array^0 }
			set_temp_variable = { dem1 = global.culture_names_array^dem1 }
		}
		localization_key = "TDA_culture_state_is_homogenous"
	}
	text = {
		trigger = {
			has_global_flag = TDA_using_super_cultures_flag

			set_temp_variable = { dem1 = FROM.super_culture_index_array^0 }
			if = {
				limit = { check_variable = { dem1 < 0 } }
				multiply_temp_variable = { dem1 = -1 }
				set_temp_variable = { dem1 = global.super_culture_names_array^dem1 }
			}
			else = {
				set_temp_variable = { dem1 = global.culture_names_array^dem1 }
			}
		}
		localization_key = "TDA_culture_state_is_homogenous"
