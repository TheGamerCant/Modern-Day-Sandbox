THIS_has_far_right_government={
	custom_trigger_tooltip={
		tooltip = THIS_has_far_right_government_tt
		OR={
			has_government = IDEOLOGY_fascist
			has_government = IDEOLOGY_ultranationalist
			has_government = IDEOLOGY_right_despotist
			has_government = IDEOLOGY_fundamentalist
			has_government = IDEOLOGY_monarchist
		}
	}
}
