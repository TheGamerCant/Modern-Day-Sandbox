scripted_map_modes = {
	TDA_population_density_map_mode = {
		top = {
			type = country
			color = {
				set_temp_variable = { red = FROM.pop_dens_r }
				set_temp_variable = { green = FROM.pop_dens_g }
				set_temp_variable = { blue = FROM.pop_dens_b }

                if = {
                    limit = { has_global_flag = TDA_pop_dens_country_view_flag }
				    set_temp_variable = { alpha = 1.0 }
				else = {
				    set_temp_variable = { alpha = 0.0 }
				}
			}
		}
		bottom = {
			type = state
			color = {
				set_temp_variable = { red = FROM.pop_dens_r }
				set_temp_variable = { green = FROM.pop_dens_g }
				set_temp_variable = { blue = FROM.pop_dens_b }

                if = {
                    limit = { has_global_flag = TDA_pop_dens_country_view_flag }
				    set_temp_variable = { alpha = 0.0 }
				else = {
				    set_temp_variable = { alpha = 1.0 }
				}
			}
		}
		far_text = none
		near_text = state
		update_daily = no
	}
}
